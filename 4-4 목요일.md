# spread syntax
```javascript
var obj = {
  a: "a",
  b: "b"
};
var str = "asdfqwer";
var arr = [11,22,33,44];

var newObj = {
  ...obj,
  ...str,
  ...arr
}
console.log(newObj);
```
```
"0":  11
"1":  22
"2":  33
"3":  44
"4":  "q"
"5":  "w"
"6":  "e"
"7":  "r"
"a":  "a"
"b":  "b"
```
